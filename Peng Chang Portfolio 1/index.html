                                     $("#btnWord").click(function () {

                                        var input = $("#txtWord").val();
                                        var userInput = "";

                                        for (var i = input.length - 1; i >= 0; i--) {
                                            userInput += input.substring(i, 1);
                                        }

                                        if (userInput == input) {
                                            $("#outputWord").text("This is Palindrome:" + ' ' + userInput);
                                        } else {
                                            $("#outputWord").text("This is not Palindrome:" + ' ' + userInput);
                                        }
                                    });

                                    //Clear button
                                    $("#btnClearPalindrome").click(function () {
                                        $("#outputWord").text("");
                                        //$("#txtWord").val('');
                                    });

                                    //Show & Hide Code
                                    $("#palindromecode").hide();
                                    $("#palindromeShow").click(function () {
                                        $("#palindromecode").toggle()
                                    });

How to FizzBuzz

Write a program that prints the numbers from 1 to 100. But for multiples of three print "Fizz" instead of the number and for the multiples of five print "Buzz". For numbers which are multiples of both three and five print "FizzBuzz".
